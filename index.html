Quick Start

// Connect to WayChain
client := waychain.Connect("https://api.waychain.org")

// Check Dox_Dev badge level
level := client.Precompile("0x13").Call("checkLevel", deployerAddr)
if level < 2 {
  return "Insufficient Dox_Dev level"
}

// Deploy contract (badge verified automatically)
tx := client.Deploy(contractCode, deployerAddr)
